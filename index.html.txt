  <script>
    // ========= CONFIG =========
    // Put your Untappd VENUE RSS URL here:
    // e.g. "https://untappd.com/v/funcleson-brew-works-home-bar/rss"
    const VENUE_RSS_URL = "https://untappd.com/v/funcleson-funkatorium/9899874/rss";

    // How often to refresh (ms)
    const REFRESH_MS = 30000;

    // ========= ELEMENTS =========
    const tickerTrack = document.getElementById("tickerTrack");
    const tickerContent = document.getElementById("tickerContent");
    const statusText = document.getElementById("statusText");

    // ========= HELPERS =========

    function timeAgo(dateString) {
      const then = new Date(dateString);
      if (isNaN(then.getTime())) return "";
      const now = new Date();
      const diffMs = now - then;
      const minutes = Math.floor(diffMs / 60000);
      const hours = Math.floor(minutes / 60);

      if (minutes < 1) return "just now";
      if (minutes < 60) return minutes + " min ago";
      if (hours < 24) return hours + " hr" + (hours !== 1 ? "s" : "") + " ago";
      const days = Math.floor(hours / 24);
      return days + " day" + (days !== 1 ? "s" : "") + " ago";
    }

    // Untappd RSS title is usually like:
    // "User is drinking a Beer Name by Brewery Name at Venue"
    function parseUntappdTitle(title) {
      if (!title) return { user: "", beer: title, brewery: "" };

      let user = "";
      let beer = "";
      let brewery = "";

      // Split "user is drinking" or "user drank"
      let userSplit = title.split(" is drinking ");
      if (userSplit.length === 1) {
        userSplit = title.split(" drank ");
      }
      if (userSplit.length > 1) {
        user = userSplit[0].trim();
        const rest = userSplit[1];

        // Split " by " for brewery
        const bySplit = rest.split(" by ");
        if (bySplit.length > 1) {
          beer = bySplit[0].replace(/^a\s+/i, "").trim();
          // Remove " at Venue" from brewery if present
          const atSplit = bySplit[1].split(" at ");
          brewery = atSplit[0].trim();
        } else {
          beer = rest.trim();
        }
      } else {
        // fallback: everything is beer
        beer = title.trim();
      }

      return { user, beer, brewery };
    }

    function buildTicker(items) {
      tickerContent.innerHTML = "";

      if (!items || !items.length) {
        const span = document.createElement("span");
        span.className = "ticker-empty";
        span.textContent = "No recent check-ins yet. Open Untappd and check in a beer at this venue.";
        tickerContent.appendChild(span);
        return;
      }

      const maxItems = Math.min(items.length, 10);
      for (let i = 0; i < maxItems; i++) {
        const item = items[i];
        const { user, beer, brewery } = parseUntappdTitle(item.title);
        const time = timeAgo(item.pubDate);

        const itemEl = document.createElement("span");
        itemEl.className = "ticker-item";

        const userEl = document.createElement("span");
        userEl.className = "ticker-user";
        userEl.textContent = user || "Someone";

        const arrowEl = document.createElement("span");
        arrowEl.textContent = "→";

        const beerEl = document.createElement("span");
        beerEl.className = "ticker-beer";
        beerEl.textContent = beer || "A beer";

        const breweryEl = document.createElement("span");
        breweryEl.className = "ticker-brewery";
        breweryEl.textContent = brewery ? "(" + brewery + ")" : "";

        const timeEl = document.createElement("span");
        timeEl.className = "ticker-time";
        timeEl.textContent = time ? "· " + time : "";

        itemEl.appendChild(userEl);
        itemEl.appendChild(arrowEl);
        itemEl.appendChild(beerEl);
        if (brewery) {
          itemEl.appendChild(breweryEl);
        }
        if (time) {
          itemEl.appendChild(timeEl);
        }

        tickerContent.appendChild(itemEl);

        if (i < maxItems - 1) {
          const sep = document.createElement("span");
          sep.className = "ticker-separator";
          sep.textContent = "◆";
          tickerContent.appendChild(sep);
        }
      }

      // Duplicate content once for smooth loop
      const clone = tickerContent.cloneNode(true);
      tickerTrack.innerHTML = "";
      tickerTrack.appendChild(tickerContent);
      tickerTrack.appendChild(clone);
    }

    async function fetchCheckins() {
      if (!VENUE_RSS_URL || VENUE_RSS_URL.includes("YOUR-VENUE-SLUG-HERE")) {
        statusText.textContent = "Configure VENUE_RSS_URL in the code.";
        return;
      }

      try {
        statusText.textContent = "Updating…";

        const rssUrl = encodeURIComponent(VENUE_RSS_URL);
        const apiUrl = `https://api.rss2json.com/v1/api.json?rss_url=${rssUrl}`;

        const res = await fetch(apiUrl);
        if (!res.ok) {
          console.error("rss2json error:", await res.text());
          statusText.textContent = "Error loading feed.";
          return;
        }

        const data = await res.json();
        const items = data.items || [];
        buildTicker(items);

        const now = new Date();
        statusText.textContent = "Last updated " + now.toLocaleTimeString();
      } catch (err) {
        console.error("Fetch error:", err);
        statusText.textContent = "Network error.";
      }
    }

    function startAutoRefresh() {
      fetchCheckins();
      setInterval(fetchCheckins, REFRESH_MS);
    }

    document.addEventListener("DOMContentLoaded", startAutoRefresh);
  </script>
